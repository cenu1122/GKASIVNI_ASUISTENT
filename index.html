import OpenAI from "openai"; const groq = new OpenAI({ apiKey:process.env.GROQ_API_KEY, baseURL:"https://api.groq.com/openai/v1" }); app.post("/api/groq",async(req,res)=>{ const completion=await groq.chat.completions.create({ model:"llama-3.3-70b-versatile", messages:[ { role:"user", content:req.body.prompt } ] }); res.json({ response:completion.choices[0].message.content }); });
